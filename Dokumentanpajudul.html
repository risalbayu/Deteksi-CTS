  <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

  <style>

    body {

      margin: 0;

      padding: 0;

    }

    #chart-container {

      width: 100vw;

      height: 60vh;

      padding: 10px;

      box-sizing: border-box;

    }

    canvas {

      width: 100% !important;

      height: 100% !important;

    }

  </style>

<body>

  <div id="chart-container">

    <canvas id="myChart"></canvas>

  </div>

  <script>

    const ctx = document.getElementById('myChart').getContext('2d');

    const chart = new Chart(ctx, {

      type: 'line',

      data: {

        labels: [],

        datasets: [{

          label: 'Suhu (°C)',

          data: [],

          borderColor: 'rgb(75, 192, 192)',

          tension: 0.1,

          pointRadius: 4,

          pointBackgroundColor: 'rgba(75,192,192,1)'

        }]

      },

      options: {

        responsive: true,

        maintainAspectRatio: false,

        animation: false,

        scales: {

          x: {

            title: { display: true, text: 'Waktu (WIB)' }

          },

          y: {

            title: { display: true, text: 'Suhu (°C)' }

          }

        }

      }

    });

    function ambilData() {

      fetch("https://myapp-b8e6f-default-rtdb.asia-southeast1.firebasedatabase.app/monitoring_tubuh/line1/history.json?orderBy=\"$key\"&limitToLast=30")

        .then(response => response.json())

        .then(data => {

          const labels = [];

          const values = [];

          for (const key in data) {

            labels.push(data[key].waktu.split(" ")[1]); // hanya jam

            values.push(data[key].suhu);

          }

          chart.data.labels = labels;

          chart.data.datasets[0].data = values;

          chart.update();

        });

    }

    ambilData();

    setInterval(ambilData, 3000);

  </script>

</body>
